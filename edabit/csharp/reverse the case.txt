using System.Text;

public class Program 
{
    public static string ReverseCase(string str) 
    {
				var sb = new StringBuilder();
				foreach (var c in str)
				{
						if (char.IsUpper(c)) {
							sb.Append(char.ToLower(c));
						}
						else {
							sb.Append(char.ToUpper(c));
						}
				}
				return sb.ToString();
    }
}
